\noindent\textbf{Definitions}
\begin{itemize}
  \item $AV$ = Approval Voting rule (selects the $k$ projects with the highest approval counts).
  \item $V$ = set of $n$ agents (voters), who make decisions i.i.d. of each other.
  \item $S$ = set of $m$ projects, indexed $1,\dots,m$.
  \item For each project $j\in S$ there is a ``true'' quality level ($\ell_j$). 
  \item For each project $j$ let
    \[
      p_j = \Pr(\text{a random voter approves project }j),
    \]
    Assume $p_j=f(\ell_j)$ for some strictly increasing $f$ -- ie, better-quality projects have larger $p_j$.
  \item For each project $j$ let
    \[
      A_j = \text{number of approvals project }j\text{ receives from the }n\text{ voters},
    \]
    so $A_j\sim\mathrm{Binomial}(n,p_j)$ and $\mathbb{E}[A_j]=np_j$.
  \item $AV(V)$ is the set of $k$ projects chosen by approval voting.
  \item Let $S^\star$ = true top-$k$ set by quality (any element of $\arg\max_{T\subseteq S, |T|=k} Q(T)$). Multiple such sets can exist, any one will suffice.
  \item Define the difference between any chosen and any non-chosen project:
    \[
      D \;:=\; \min_{\;j\in S^\star,\;\ell\notin S^\star} (p_j - p_\ell).
    \]
    Assume $D>0$ (if $D=0$, there are exact ties in approval probabilities).
\end{itemize}

\noindent\textbf{Proposition.}
\[
  \Pr\big( Q(AV(V)) \in \arg\max_{T\subseteq S,|T|=k} Q(T) \big)
  \;\ge\; 1 - m^2 \exp\!\Big(-\tfrac{1}{2} n D^2\Big).
\]\\


\noindent\textbf{Proof.}

\textbf{Mistake Case}  
AV makes a mistake only if there exists some ordered pair $(j,\ell)$ with $j\in S^\star,\ \ell\notin S^\star$ such that $A_\ell \ge A_j$.

Hence, the lower bound probability of such a mistake is defined as the probability that there exists $j\in S^\star$ and $\ell\notin S^\star$ with $A_\ell \ge A_j$.

\textbf{Fix one pair $(j,\ell)$.}  
Fix $j\in S^\star$ and $\ell\notin S^\star$. Consider the difference
\[
  D := A_j - A_\ell.
\]
We have $\mathbb{E}[D] = n(p_j - p_\ell) \ge nD$ by definition of $D$.

\textbf{$D$ as a sum of independent bounded variables.}  
For voter $i=1,\dots,n$ define
\[
  X_i := \mathbf{1}\{\text{voter }i\text{ approves }j\} - \mathbf{1}\{\text{voter }i\text{ approves }\ell\}.
\]
Then $X_i\in\{-1,0,1\}$, the $X_i$ are independent across $i$, and
\[
  D = \sum_{i=1}^n X_i,\qquad \mathbb{E}[X_i]=p_j-p_\ell.
\]

\textbf{Hoeffding's Inequality.}  
Each $X_i$ is bounded in $[-1,1]$. Hoeffding's inequality yields, for any $t>0$,
\[
  \Pr\Big( \sum_{i=1}^n X_i - \mathbb{E}\big[\sum_{i=1}^n X_i\big] \le -t \Big)
  \le \exp\!\Big(-\frac{t^2}{2n}\Big).
\]
Take $t=\mathbb{E}[D]\ge nD$. Then
\[
  \Pr(D \le 0)
  = \Pr\Big(\sum_{i=1}^n X_i - \mathbb{E}[D] \le -\mathbb{E}[D]\Big)
\]
\[
  \le \exp\!\Big(-\frac{(\mathbb{E}[D])^2}{2n}\Big)
  \le \exp\!\Big(-\tfrac{1}{2} n D^2\Big).
\]
Thus for this fixed ordered pair $(j,\ell)$ the probability that the outside project $\ell$ ties or beats the inside project $j$ is at most $\exp(-\tfrac12 nD^2)$.

\textbf{Union bound.}  
AV makes a mistake only if there exists some ordered pair $(j,\ell)$ with $j\in S^\star,\ \ell\notin S^\star$ such that $A_\ell \ge A_j$. The number of such ordered pairs is $|S^\star|\cdot (m-|S^\star|)\le m^2$. By the union bound,
\[
  \Pr\big(S_{\mathrm{AV}} \ne S^\star\big)
  \le \sum_{j\in S^\star}\sum_{\ell\notin S^\star} \Pr(A_\ell \ge A_j)
  \le m^2 \exp\!\Big(-\tfrac{1}{2} n D^2\Big).
\]

Therefore
\[
  \Pr\big(S_{\mathrm{AV}} = S^\star\big)
  \ge 1 - m^2 \exp\!\Big(-\tfrac{1}{2} n D^2\Big),
\]
and the RHS tends to 1 exponentially fast as $n\to\infty$. \(\square\)
